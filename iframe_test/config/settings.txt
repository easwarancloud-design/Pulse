from pydantic_settings import BaseSettings
from typing import Optional

class Settings(BaseSettings):
    """Application settings from environment variables"""
    
    # Application
    APP_NAME: str = "Pulse Conversations API"
    APP_VERSION: str = "1.0.0"
    DEBUG: bool = False
    
    # MySQL Database
    MYSQL_HOST: str = "localhost"
    MYSQL_PORT: int = 3306
    MYSQL_USER: str = "pulse_app"
    MYSQL_PASSWORD: str = "your_secure_password_here"
    MYSQL_DATABASE: str = "pulse_conversations"
    MYSQL_POOL_SIZE: int = 10
    MYSQL_MAX_OVERFLOW: int = 20
    
    # Redis
    REDIS_HOST: str = "localhost"
    REDIS_PORT: int = 6379
    REDIS_DB: int = 0
    REDIS_PASSWORD: Optional[str] = None
    REDIS_POOL_SIZE: int = 20
    REDIS_TTL_SECONDS: int = 900  # 15 minutes
    
    # CORS
    CORS_ORIGINS: list = ["http://localhost:3000", "http://localhost:3002"]
    
    # API Settings
    API_V1_PREFIX: str = "/api/v1"
    
    class Config:
        env_file = ".env"
        case_sensitive = True

settings = Settings()
